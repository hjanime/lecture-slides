\foreach \angle in {0,1,...,359}{
    \begin{tikzpicture}
        % Axis
        \draw[thick,->,black] (-3,0)--(3,0) node[below] {$x$}; % x axis
        \draw[thick,->,black] (0,-3)--(0,3) node[left] {$y$}; % y axis
        \draw[black,thin] (0,0) circle (2.5cm);
        \node[black,below] at (2.6,0) {1};
        \node[black,above] at (0.1,-2.5) {1};
        %
        \draw[ultra thick,cyan] (0,0) -- (0,0 |- \angle:2.5cm); % UpOn x axis
        \draw[ultra thick,orange] (0,0) -- (\angle:2.5cm |- 0,0); % UpOn y axis
        %
        \draw[densely dotted,cyan] (\angle:2.5cm) -- (\angle:2.5cm |- 0,0); % vertical line
        \draw[densely dotted,orange] (\angle:2.5cm) -- (0,0 |- \angle:2.5cm); % horizontal line
        \draw[ultra thick,red,->,rotate=\angle] (0,0) -- (2.5,0);
        \node[red,orange,right] at (0,-3.5)
        {\footnotesize$\cos(\angle^{\mathrm{o}}) = \pgfmathcos{\angle}\pgfmathresult$};
        \node[red,cyan,right] at (0,-3.1)
        {\footnotesize$\sin(\angle^{\mathrm{o}}) = \pgfmathsin{\angle}\pgfmathresult$};
    \end{tikzpicture}
}
